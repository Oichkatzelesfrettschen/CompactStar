223 void Print(const int n_lines = 5) const
224 {
225 // Guard: no data
226 if (eps.empty() || pre.empty() || rho.empty())
227 {
228 std::cout << "\n[EOS::Print] Warning: table is empty — nothing to print.\n";
229 return;
230 }
231
232 std::cout << "\n *-------------------------------------------* \n";
233 std::cout << " | " << eps_label << " | " << pre_label
234 << " | " << rho_label << " |\n";
235 std::cout << " *-------------------------------------------* \n";
236
237 // Determine number of rows to print
238 size_t total = eps.size();
239 size_t limit = total;
240
241 if (n_lines >= 0)
242 {
243 limit = std::min<size_t>(n_lines, total);
244 } // if n_lines < 0, print all rows
245
246 // Loop over selected rows
247 for (size_t i = 0; i < limit; i++)
248 {
249 std::cout << " | "
250 << std::scientific << std::setprecision(6)
251 << eps[i] << " | "
252 << pre[i] << " | "
253 << rho[i] << "\n";
254 }
255
256 // If we didn’t print all lines, indicate how many were omitted
257 if (limit < total)
258 {
259 std::cout << " | ... (" << (total - limit)
260 << " more rows omitted) ...\n";
261 }
262
263 std::cout << " *-------------------------------------------* \n";
264 }
265
274 void PrintSummary() const
275 {
276 std::cout << "\n================ EOS Table Summary ================\n";
277
278 if (eps.empty() || pre.empty() || rho.empty())
279 {
280 std::cout << "[EOS::PrintSummary] Table is empty.\n";
281 std::cout << "===================================================\n";
282 return;
283 }
284
285 const size_t n = eps.size();
286
287 auto [eps_min, eps_max] = std::minmax_element(eps.begin(), eps.end());
288 auto [pre_min, pre_max] = std::minmax_element(pre.begin(), pre.end());
289 auto [rho_min, rho_max] = std::minmax_element(rho.begin(), rho.end());
290
291 std::cout << std::scientific << std::setprecision(6);
292 std::cout << " # of rows: " << n << "\n";
293 std::cout << " Columns: " << eps_label << " | " << pre_label << " | " << rho_label << "\n";
294 std::cout << " ε (energy density): [" << *eps_min << ", " << *eps_max << "]\n";
295 std::cout << " P (pressure): [" << *pre_min << ", " << *pre_max << "]\n";
296 std::cout << " ρ (baryon density): [" << *rho_min << ", " << *rho_max << "]\n";
297
298 if (!extra_labels.empty())
299 {
300 std::cout << " Additional species: ";
301 for (size_t i = 0; i < extra_labels.size(); ++i)
302 {
303 std::cout << extra_labels[i];
304 if (i + 1 < extra_labels.size())
305 std::cout << ", ";
306 }
307 std::cout << "\n";
308 }
309
310 std::cout << "===================================================\n";
311 }

316struct TOVPoint
317{
318 // [Nov 1, 2025] m is in solar mass units
319 double r, m, nu_der, nu, p, e, rho;
320 std::vector<double> rho_i;
321
322 TOVPoint(const double &in_r, const double &in_m,
323 const double &in_nu_der, const double &in_nu,
324 const double &in_p, const double &in_e,
325 const double &in_rho, const std::vector<double> &in_rho_i)
326 : r(in_r), m(in_m), nu_der(in_nu_der), nu(in_nu),
327 p(in_p), e(in_e), rho(in_rho),
328 rho_i(in_rho_i)
329 {
330 }
331
332 std::string Str() const
333 {
334 std::stringstream ss;
335 char tmp[200];
336 snprintf(tmp, sizeof(tmp), "%.8e\t %.8e\t %.8e\t %.8e\t %.8e\t %.8e\t %.8e",
337 r, m, nu_der, nu, p, e, rho);
338 ss << tmp;
339 for (double tmp_rho : rho_i)
340 {
341 snprintf(tmp, sizeof(tmp), "\t %.8e", tmp_rho);
342 ss << tmp;
343 }
344
345 return ss.str();
346 }
