59 : x(in_x), y(in_y), z(in_z) {}

61 bool operator==(const Coord3D& rhs) const
62 {
63 return ( x == rhs.x && y == rhs.y && z == rhs.z ) ;
64 }
65
66 bool operator>(const Coord3D& rhs) const
67 {
68 return ( XYDist2({0,0,0}) > rhs.XYDist2({0,0,0})) ;

70

72 {
73 return ( XYDist2({0,0,0}) < rhs.XYDist2({0,0,0})) ;
74 }
75
76 double XYDist2(const Coord3D& rhs) const
77 {

117struct XYZVector
118{
119 double X, Y, Z ;
120
121 // .....................................
122 // Operators
123 bool operator==(const XYZVector& other)
124 { return X == other.X && Y == other.Y && Y == other.Y; }
125
126 XYZVector operator+(const XYZVector& other)
127 { return {X + other.X, Y + other.Y, Z + other.Z}; }
128
129 XYZVector operator-(const XYZVector& other)
130 { return {X - other.X, Y - other.Y, Z - other.Z}; }
131
132 XYZVector operator*(const int& other)
133 { return {X * other, Y * other, Z * other}; }
134
135 XYZVector operator*(const double& other)
136 { return {X * other, Y * other, Z * other}; }
137
138 XYZVector operator-()
139 { return {-X, -Y, -Z}; }
140 // .....................................
142 double mag2() const { return X*X + Y*Y + Z*Z; }
143};
