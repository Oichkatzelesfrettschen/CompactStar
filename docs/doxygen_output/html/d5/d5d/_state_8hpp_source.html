137 virtual void SanityCheck() const
138 {
139 PROFILE_FUNCTION();
140
141 const std::size_t n = Size();
142 const double *y = Data();
143
144 if (!y || n == 0)
145 {
146 Z_LOG_WARNING(std::string("SanityCheck: empty state in ") + Name());
147 return;
148 }
149
150 bool has_nan = false, has_inf = false;
151 for (std::size_t i = 0; i < n; ++i)
152 {
153 if (std::isnan(y[i]))
154 {
155 has_nan = true;
156 break;
157 }
158 if (!std::isfinite(y[i]))
159 {
160 has_inf = true;
161 break;
162 }
163 }
164
165 if (has_nan)
166 Z_LOG_ERROR(std::string("SanityCheck[") + Name() + "]: NaN detected");
167 if (has_inf)
168 Z_LOG_ERROR(std::string("SanityCheck[") + Name() + "]: Inf detected");
169 if (!has_nan && !has_inf)
170 Z_LOG_INFO(std::string("SanityCheck[") + Name() + "]: OK (" +
171 std::to_string(n) + " DOFs)");
172 }
