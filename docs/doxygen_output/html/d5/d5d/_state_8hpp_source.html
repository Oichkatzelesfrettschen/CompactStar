181 virtual void SanityCheck() const
182 {
183 PROFILE_FUNCTION();
184
185 const std::size_t n = Size();
186 const double *y = Data();
187
188 if (!y || n == 0)
189 {
190 Z_LOG_WARNING(std::string("SanityCheck: empty state in ") + Name());
191 return;
192 }
193
194 bool has_nan = false, has_inf = false;
195 for (std::size_t i = 0; i < n; ++i)
196 {
197 if (std::isnan(y[i]))
198 {
199 has_nan = true;
200 break;
201 }
202 if (!std::isfinite(y[i]))
203 {
204 has_inf = true;
205 break;
206 }
207 }
208
209 if (has_nan)
210 Z_LOG_ERROR(std::string("SanityCheck[") + Name() + "]: NaN detected");
211 if (has_inf)
212 Z_LOG_ERROR(std::string("SanityCheck[") + Name() + "]: Inf detected");
213 if (!has_nan && !has_inf)
214 Z_LOG_INFO(std::string("SanityCheck[") + Name() + "]: OK (" +
215 std::to_string(n) + " DOFs)");
216 }
