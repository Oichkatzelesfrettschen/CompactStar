414struct Row
415{
416 public:
417 std::vector<double> vals ;
418 char delim = '\t' ;
419
420 // Precision for printing (double) numbers
421 int precision = 8 ;
422
423 Row() {}
424
425 Row(const std::vector<double>& in_vals )
426 {
427 vals = in_vals ;
428 }
429
430 std::string Str() const
431 {
432 std::stringstream ss;
433
434 // [Apr 12, 2023] : Delim issue was resolved.
435 for (size_t i = 0; i < vals.size() - 1 ; i++)
436 {
437 char tmp[150] ;
438 snprintf(tmp, sizeof(tmp), "%-*.*e%c ", 7 + precision,
439 precision, vals[i], delim) ;
440 ss << tmp ;
441 }
442
443
444
445 // Last column
446 char tmp[150] ;
447 snprintf(tmp, sizeof(tmp), "%-*.*e ", 7 + precision,
448 precision, vals[vals.size() - 1]) ;
449 ss << tmp ;
450
451 return ss.str() ;
452 }
