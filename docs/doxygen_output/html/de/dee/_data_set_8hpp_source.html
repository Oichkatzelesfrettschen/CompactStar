391struct Row
392{
393 public:
394 std::vector<double> vals;
395 char delim = '\t';
396
397 // Precision for printing (double) numbers
398 int precision = 8;
399
400 Row() {}
401
402 Row(const std::vector<double> &in_vals)
403 {
404 vals = in_vals;
405 }
406
407 std::string Str() const
408 {
409 std::stringstream ss;
410
411 // [Apr 12, 2023] : Delim issue was resolved.
412 for (size_t i = 0; i < vals.size() - 1; i++)
413 {
414 char tmp[150];
415 snprintf(tmp, sizeof(tmp), "%-*.*e%c ", 7 + precision,
416 precision, vals[i], delim);
417 ss << tmp;
418 }
419
420 // Last column
421 char tmp[150];
422 snprintf(tmp, sizeof(tmp), "%-*.*e ", 7 + precision,
423 precision, vals[vals.size() - 1]);
424 ss << tmp;
425
426 return ss.str();
427 }
428};
