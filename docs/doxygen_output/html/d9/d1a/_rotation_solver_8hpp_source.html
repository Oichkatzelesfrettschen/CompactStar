69// Added on Aug 6, 2020
70// omega_bar_c: The initial value for bar{omega} at r = 0
71// Note that bar{omega}(r) = Omega - omega(r)
72//
73// m : mass of the star
74// r : radius
75// J : total angular momentum of the star
76// Omega : angular velocity of the star
77//
78struct OmegaSeqPoint
79{
80 double omega_bar_c, m, r, J, Omega;
81
82 OmegaSeqPoint(const double &in_oc, const double &in_m,
83 const double &in_r, const double &in_J, const double &in_Omega)
84 : omega_bar_c(in_oc), m(in_m), r(in_r), J(in_J), Omega(in_Omega)
85 {
86 }
87
88 std::string Str() const
89 {
90 std::stringstream ss;
91 char tmp[200];
92 snprintf(tmp, sizeof(tmp), "%.8e\t %.8e\t %.8e\t %.8e\t %.8e", omega_bar_c, m, r, J, Omega);
93 ss << tmp;
94
95 return ss.str();
96 }
97};
