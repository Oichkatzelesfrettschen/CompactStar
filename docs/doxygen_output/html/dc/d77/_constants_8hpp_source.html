386struct Element
387{
388 std::string sym = "" ;
389 double Z = 0 ; // Atomic number
390 double A = 0 ; // Mass number, Standard atomic weight
391 double I = 0 ; // DM capture rate in the Sun
392 double v2 = 0; // Typical escape velocity squared in the Sun
393
394 Element() {}
395 Element(const std::string& in_sym, double in_Z, double in_A)
396 : sym(in_sym), Z(in_Z), A(in_A) {}
397 Element(const std::string& in_sym, double in_Z, double in_A, double in_I, double in_v2)
398 : sym(in_sym), Z(in_Z), A(in_A), I(in_I), v2(in_v2) {}
