503 Zaki::Vector::DataColumn GetSpecies(const std::string &label) const
504 {
505 int li = SpeciesLocalIndex(label);
506 if (li < 0 || static_cast<std::size_t>(li) >= species_idx.size())
507 throw std::out_of_range("Unknown species label: " + label);
508
509 int col_idx = species_idx[static_cast<std::size_t>(li)];
510 if (!IsValidColumnIndex(col_idx))
