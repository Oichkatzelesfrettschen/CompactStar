501 Zaki::Vector::DataColumn GetSpecies(const std::string &label) const
502 {
503 int li = SpeciesLocalIndex(label);
504 if (li < 0 || static_cast<std::size_t>(li) >= species_idx.size())
505 throw std::out_of_range("Unknown species label: " + label);
506
507 int col_idx = species_idx[static_cast<std::size_t>(li)];
