218 virtual void SanityCheck() const
219 {
220 PROFILE_FUNCTION();
221 const std::size_t n = Size();
222 const double *y = Data();
223 if (!y || n == 0)
224 {
225 Z_LOG_WARNING(std::string("State::SanityCheck: empty state in ") + Name());
226 return;
227 }
228
229 bool has_nan = false, has_inf = false;
230 for (std::size_t i = 0; i < n; ++i)
231 {
232 if (std::isnan(y[i]))
233 {
234 has_nan = true;
235 break;
236 }
237 if (!std::isfinite(y[i]))
238 {
239 has_inf = true;
240 break;
241 }
242 }
243 if (has_nan)
244 Z_LOG_ERROR(std::string("SanityCheck[") + Name() + "]: NaN detected.");
245 if (has_inf)
246 Z_LOG_ERROR(std::string("SanityCheck[") + Name() + "]: Inf detected.");
247 if (!has_nan && !has_inf)
248 Z_LOG_INFO(std::string("SanityCheck[") + Name() + "]: OK (n=" + std::to_string(n) + ").");
249 }
