64class MixedSeqPoint
65{
66 public:
67 SeqPoint v, d;
68 size_t v_idx = 0, d_idx = 0;
69
71 MixedSeqPoint() {};
72
80 MixedSeqPoint(const size_t &in_v_idx,
81 const SeqPoint &in_v,
82 const size_t &in_d_idx,
83 const SeqPoint &in_d)
84 : v(in_v), d(in_d), v_idx(in_v_idx),
85 d_idx(in_d_idx)
86 {
87 }
88
92 MixedSeqPoint(const size_t &in_v_idx,
93 const size_t &in_d_idx,
94 const std::vector<double> &in_seq_row)
95 : v_idx(in_v_idx), d_idx(in_d_idx)
96 {
97 if (in_seq_row.size() != 12)
98 Z_LOG_ERROR("The mixed sequence data point is incomplete.");
99 else
100 {
101 v = {in_seq_row[0], in_seq_row[1], in_seq_row[2],
102 in_seq_row[3], in_seq_row[4], in_seq_row[5]};
103 d = {in_seq_row[6], in_seq_row[7], in_seq_row[8],
104 in_seq_row[9], in_seq_row[10], in_seq_row[11]};
105 }
106 }
107
114 std::string Str() const
115 {
116 // May 10, 2022: For some reason
117 // %lu and %zu specifiers don't
118 // print zeros of v_idx and d_idx!
119 // Fixed: Removed "." in "%5.lu".
120 std::stringstream ss;
121 char tmp[75];
122 snprintf(tmp, sizeof(tmp), "%5lu\t %5lu\t ", v_idx, d_idx);
123 // std::cout << "\n\t\t" << tmp << "\n" ;
124 ss << tmp << v.Str() << "\t " << d.Str();
125 return ss.str();
126 }
127
129 void Reset()
130 {
131 v_idx = 0;
132 d_idx = 0;
133 v.clear();
134 d.clear();
135 }
136
137 //..................................................
138 // Operator overloading
139 // Addition (+)
143 MixedSeqPoint operator+(const MixedSeqPoint &seq) const
144 {
145 MixedSeqPoint out_seq(0, v + seq.v, 0, d + seq.d);
146 return out_seq;
147 }
148
149 // Multiplication (*)
153 MixedSeqPoint operator*(const double &num) const
154 {
155 MixedSeqPoint out_seq(0, v * num, 0, d * num);
156 return out_seq;
157 }
158 //..................................................
159};
