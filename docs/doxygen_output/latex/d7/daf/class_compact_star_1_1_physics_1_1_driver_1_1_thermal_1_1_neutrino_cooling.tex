\doxysubsection{Detailed Description}
Adds a neutrino-\/cooling contribution to the thermal RHS (redshifted frame). 

{\bfseries{Depends on\+:}} \mbox{\hyperlink{namespace_compact_star_1_1_physics_1_1_thermal}{Thermal}} {\bfseries{Updates\+:}} \mbox{\hyperlink{namespace_compact_star_1_1_physics_1_1_thermal}{Thermal}}\hypertarget{class_compact_star_1_1_physics_1_1_driver_1_1_thermal_1_1_neutrino_cooling_autotoc_md29}{}\doxyparagraph{What this driver does}\label{class_compact_star_1_1_physics_1_1_driver_1_1_thermal_1_1_neutrino_cooling_autotoc_md29}

\begin{DoxyEnumerate}
\item Reads the evolved thermal DOF (T\+\_\+\textbackslash{}infty) from the Thermal\+State in {\ttfamily Y}.
\item Computes derived neutrino cooling quantities through {\ttfamily \mbox{\hyperlink{_neutrino_cooling___details_8hpp_a017b32db80be303c7f5724ca4770e18f}{Detail\+::\+Compute\+Derived()}}}.
\item Accumulates the derived (dx/dt = d/dt\textbackslash{}ln(T\+\_\+\textbackslash{}infty/\+T\+\_\+\{\textbackslash{}rm ref\})) into the RHS.
\end{DoxyEnumerate}\hypertarget{class_compact_star_1_1_physics_1_1_driver_1_1_thermal_1_1_neutrino_cooling_autotoc_md30}{}\doxyparagraph{What this driver does not do (yet)}\label{class_compact_star_1_1_physics_1_1_driver_1_1_thermal_1_1_neutrino_cooling_autotoc_md30}

\begin{DoxyItemize}
\item It does not implement the full microphysics emissivity set by default.
\item It does not compute (C\+\_\+\{\textbackslash{}rm eff\}) from EOS/microphysics unless you wire it.
\item It does not modify structure (TOV/geometry profiles).
\end{DoxyItemize}\hypertarget{class_compact_star_1_1_physics_1_1_driver_1_1_thermal_1_1_neutrino_cooling_autotoc_md31}{}\doxyparagraph{Determinism contract}\label{class_compact_star_1_1_physics_1_1_driver_1_1_thermal_1_1_neutrino_cooling_autotoc_md31}

\begin{DoxyItemize}
\item The driver is pure with respect to {\ttfamily Y} and {\ttfamily ctx} (read-\/only).
\item For fixed {\ttfamily (t, Y, ctx)} it produces deterministic output. 
\end{DoxyItemize}
